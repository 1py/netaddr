Different way of formatting IPv6 addresses.

{{{

>>> ip = IPAddress(0, 6)
>>> str(ip)
'::'

>>> ip = IPAddress(0, 6, dialect=ipv6_compact)
>>> str(ip)
'::'

>>> ip = IPAddress(0, 6, dialect=ipv6_full)
>>> str(ip)
'0:0:0:0:0:0:0:0'

>>> ip = IPAddress(0, 6, dialect=ipv6_verbose)
>>> str(ip)
'0000:0000:0000:0000:0000:0000:0000:0000'

>>> ip = IPAddress('1080::8:800:200c:417a')
>>> str(ip)
'1080::8:800:200c:417a'

>>> ip = IPAddress('1080::8:800:200c:417a', dialect=ipv6_compact)
>>> str(ip)
'1080::8:800:200c:417a'

>>> ip = IPAddress('1080::8:800:200c:417a', dialect=ipv6_full)
>>> str(ip)
'1080:0:0:0:8:800:200c:417a'

>>> ip = IPAddress('1080::8:800:200c:417a', dialect=ipv6_verbose)
>>> str(ip)
'1080:0000:0000:0000:0008:0800:200c:417a'

}}}
